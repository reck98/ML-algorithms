from scipy.stats import chi2_contingency

alpha = 0.05

categorical_cols =['is_female',
 'is_smoker',
 'is_southeast',
 'is_southwest',
 'is_northwest',
 'is_northeast',
 'is_overweight',
 'is_obese',
 'is_underweight',
 'is_healthy',
]

keep_cols = []


## H0 --> There is no coorelation 

for col in categorical_cols:
    contingency_table = pd.crosstab(df_cleaned[col], df_cleaned['charges_bin'])
    chi2, p, dof, expected = chi2_contingency(contingency_table)

    if p < alpha:
        print(f"{col}: p={p:.4f} → Reject H0 (keep)")
        keep_cols.append(col)
    else:
        print(f"{col}: p={p:.4f} → Fail to reject H0 (drop)")
